  .icon-#{$name} {

  width: $size;
  height: $size;

@function for computing expressions

@function pow($base, $exp) {
  $res: 1;
  @for $_ from 1 through $exp {
    $res: $res * $base;
  }
  @return $res;
}

.bar {
  margin: pow(2, 6) * 1px;
}

any number of arguments

@function sum($nums...) {
  $sum: 0;
  @each $n in $nums {
    $sum: $sum + $n;
  }
  @return $sum;
}

.small {
  width: sum(10px, 50px, 30px);
}

Debugging: @error and @warn

@mixin shadow($opacity, $blur, $offset) {
  @if $opacity <= 0 or $opacity > 1 {
    @error "Opacity should be within 0 and 1, got #{$opacity}";
  }

  @if $offset == 0 {
    @warn "box shadow offset was set at 0";
  }

  box-shadow: $offset $offset $blur rgba(0, 0, 0, $opacity);
}
